import socket,os


def main():
    global data, s
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.bind(('127.0.0.1',10001))
    s.listen(5)
    print("等待连接中……")
    while True:
        conn,addr = s.accept()
        while True:
            print('收到命令')
            data = conn.recv(1024)
            cmd = os.popen(data.decode())
            os_result = cmd.read()
            conn.sendall(os_result.encode('utf8'))


if __name__ == '__main__':
    main()
